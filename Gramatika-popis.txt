while_statement : tWhile tLPar condition tRPar tLCurl statements tRCurl

functionDeclaration : tFunction tFuncName tLPar arguments tRPar tColon type tRCurl statements tLCurl

arguments :  
                |variable variables

functionCall : tFuncName tLPar parameters tRPar
                  
parameters : 
                |parameter parameters2

parameters2 : 
                 |tComma parameter parameters2
                 
parameter : factor

statements :  
                 |statement statements

statement :  expression statement2 tSemicolon
                |if_statement
                |while_statement
                |functionCall tSemicolon
                |tReturn expression tSemiColon

statement2 : tAssign expression
		    |
		    
condition :  expression condition2

condition2 :  
                 |relational_operators expression

expression :  term expression2
                 |
                 
expression2 :  tPlus expression
                  |tMinus expression
                  |

term :  const term2
           |variable term2
	   |factor term2 // potencialne jenom tohle
           
term2 :  tMul term // factor
            |tDiv term // factor
            |tConcat term //factor
            |
             
factor : variable
           |const

variables :  
                |tComma variable variables

variable : type tIdentifier | tIdentifier
           
const : tInt | tReal | tReal2 | tInt2 | tNull | tLiteral

type :  tNullTypeInt | tNullTypeFloat | tNullTypeString
	     |tTypeInt | tTypeFloat | tTypeString
	     |tVoid
            
relational_operators :  tLess | tLessEq | tMore
                            |tMoreEq | tIdentical | tNotIdentical
